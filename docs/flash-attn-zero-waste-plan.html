"""TE-based zero-waste Flash Attention backend for prefix-sharing. Uses Transformer Engine's DotProductAttention with post_scale_bias to implement block-causal masking without Q padding. """ from __future__ import annotations from typing import Any import torch from prefix_sharing.backends.base import BackendCapabilities from prefix_sharing.backends.torch_ref import TorchReferenceBackend, _split_packed from prefix_sharing.core.planner import PrefixSharingPlan def build_block_causal_bias( plan: PrefixSharingPlan, num_heads: int, device: torch.device, dtype: torch.dtype, ) -> torch.Tensor: """Construct per-sequence block-causal bias tensor. Shape: [batch_size, 1, max_seqlen_q, max_seqlen_kv] Values: 0.0 for visible, -inf for masked. """ B = plan.batch_size max_q = plan.max_seqlen_q max_kv = plan.max_seqlen_kv bias = torch.full((B, 1, max_q, max_kv), float('-inf'), device=device, dtype=dtype) for i in range(B): q_len = plan.kept_lengths_q[i] kv_len = plan.expanded_lengths_kv[i] if not plan.is_reuser(i): # Provider/standalone: standard lower-triangular causal for qi in range(q_len): bias[i, 0, qi, :qi + 1] = 0.0 else: # Reuser: prefix columns all visible + suffix causal p = plan.prefix_lens[i] for qi in range(q_len): bias[i, 0, qi, :p] = 0.0 # 前缀全可见 bias[i, 0, qi, p:p + qi + 1] = 0.0 # 后缀 causal return bias def _make_te_attention(num_heads, kv_channels, softmax_scale): """Create a TE DotProductAttention instance.""" import transformer_engine as te return te.pytorch.DotProductAttention( num_attention_heads=num_heads, kv_channels=kv_channels, attention_dropout=0.0, attn_mask_type='padding', # THD 需要 padding 类 mask softmax_scale=softmax_scale, ) class FlashAttentionBackend(TorchReferenceBackend): capabilities = BackendCapabilities( name="flash_attn", supports_cpu=False, supports_cuda=True, supports_cann=False, supports_different_q_kv_lengths=True, supports_prefix_last_restore=True, ) # build_kv() 完全继承 TorchReferenceBackend def attention( self, query: Any, key: Any, value: Any, plan: PrefixSharingPlan, **kwargs: Any, ) -> Any: num_heads = query.shape[1] head_dim = query.shape[2] softmax_scale = head_dim ** -0.5 # Step 1: Construct block-causal bias bias = build_block_causal_bias( plan, num_heads=num_heads, device=query.device, dtype=query.dtype, ) # Step 2: Prepare cu_seqlens tensors cu_q = torch.tensor(plan.cu_seqlens_q, dtype=torch.int32, device=query.device) cu_kv = torch.tensor(plan.cu_seqlens_kv, dtype=torch.int32, device=key.device) # Step 3: Create TE attention module (cached per layer in practice) te_attn = _make_te_attention(num_heads, head_dim, softmax_scale) # Step 4: Call TE with bias + THD packed params output = te_attn( query.unsqueeze(1), # [T, 1, H, D] → TE expects batch dim key.unsqueeze(1), value.unsqueeze(1), attention_mask='padding', core_attention_bias_type='post_scale_bias', core_attention_bias=bias, qkv_format='thd', cu_seqlens_q=cu_q, cu_seqlens_k=cu_kv, max_seqlen_q=plan.max_seqlen_q, max_seqlen_k=plan.max_seqlen_kv, ).squeeze(1) return output
关键: Q 不 pad，cu_seqlens_q 使用 kept_lengths_q（suffix-only），cu_seqlens_k 使用 expanded_lengths_kv（full）

8. 梯度安全性

操作梯度行为正确性
build_kv (继承 torch_ref)torch.cat 梯度拆分，Provider KV 累加与当前一致
build_block_causal_biasbias 是整数运算，不参与梯度无影响
TE DotProductAttention原生 autograd，支持 backwardTE 保证
expanded KV 中的 Provider KV梯度通过 TE backward → cat → Provider等价于 torch_ref
Q (不 pad)只有 suffix Q 参与计算，梯度完整无 pad = 无额外问题
比 Q Padding 方案更简单：没有 pad 向量，没有 trim 切片，梯度路径更直接
