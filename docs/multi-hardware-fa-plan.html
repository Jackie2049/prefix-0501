3. NPU Backend 详细代码

prefix_sharing/backends/npu_flash.py ★ 新建 (~100 行)
"""NPU (Huawei Ascend) Flash Attention backend via npu_fusion_attention. Uses npu_fusion_attention with TND layout, actual_seq_qlen/kvlen, and atten_mask for block-causal masking. Zero FLOPs waste. """ from __future__ import annotations from typing import Any import torch from prefix_sharing.backends.base import BackendCapabilities from prefix_sharing.backends.torch_ref import TorchReferenceBackend from prefix_sharing.backends.mask import build_block_causal_mask from prefix_sharing.core.planner import PrefixSharingPlan class NpuFlashBackend(TorchReferenceBackend): """NPU-optimized backend using npu_fusion_attention.""" capabilities = BackendCapabilities( name="npu_flash", supports_cpu=False, supports_cuda=False, supports_cann=True, supports_different_q_kv_lengths=True, supports_prefix_last_restore=True, ) # build_kv() 完全继承 TorchReferenceBackend def attention( self, query: Any, # [total_q, num_heads, head_dim] TND key: Any, # [total_kv, num_heads, head_dim] TND value: Any, plan: PrefixSharingPlan, **kwargs: Any, ) -> Any: from mindspeed.ops import npu_fusion_attention num_heads = query.shape[1] head_dim = query.shape[2] scale = head_dim ** -0.5 # Step 1: 统一构造 block-causal mask atten_mask = build_block_causal_mask(plan, device=query.device) # Step 2: cu_seqlens → Python list (npu_fusion_attention 要求 list) cu_q = plan.cu_seqlens_q cu_kv = plan.cu_seqlens_kv if isinstance(cu_q, torch.Tensor): cu_q = cu_q.tolist() if isinstance(cu_kv, torch.Tensor): cu_kv = cu_kv.tolist() # Step 3: 调用 npu_fusion_attention (TND packed format) # sparse_mode=0: 使用 atten_mask 参数控制 mask output, _, _, _, _, _, _ = npu_fusion_attention( query, key, value, num_heads, 'TND', # packed layout atten_mask=atten_mask, # BoolTensor, True=masked scale=scale, keep_prob=1.0, # no dropout during training sparse_mode=0, # use atten_mask actual_seq_qlen=cu_q, # list: [0, q1, q1+q2, ...] actual_seq_kvlen=cu_kv, # list: [0, kv1, kv1+kv2, ...] ) return output

4. 新建 backends/mask.py — 统一 mask 构造（共享）

prefix_sharing/backends/mask.py ★ 新建 (~60 行，三个 backend 共享)
"""Platform-agnostic block-causal mask construction for prefix-sharing. Output: BoolTensor [max_seqlen_q, max_seqlen_kv] Semantics: True = position is MASKED (not visible), False = visible This matches npu_fusion_attention's atten_mask convention directly. CUDA/TE converts ~mask to float bias; Fallback uses it with masked_fill. """ import torch from prefix_sharing.core.planner import PrefixSharingPlan def build_block_causal_mask( plan: PrefixSharingPlan, device: torch.device = torch.device('cpu'), ) -> torch.Tensor: """Build per-sequence block-causal attention mask. For each sequence i: - Provider/standalone: standard upper-triangular causal mask - Reuser: prefix columns all-visible + suffix causal """ max_q = plan.max_seqlen_q max_kv = plan.max_seqlen_kv # Start with all True (all masked) mask = torch.ones(max_q, max_kv, dtype=torch.bool, device=device) for i in range(plan.batch_size): q_len = plan.kept_lengths_q[i] kv_len = plan.expanded_lengths_kv[i] if not plan.is_reuser(i): # Provider: standard causal → upper triangle = True # Only need to set the lower triangle (including diag) to False for qi in range(q_len): mask[qi, :qi + 1] = False else: p = plan.prefix_lens[i] # Reuser: prefix all visible + suffix causal for qi in range(q_len): mask[qi, :p] = False # prefix columns visible mask[qi, p:p + qi + 1] = False # suffix causal return mask def mask_to_te_bias(mask: torch.Tensor, dtype: torch.dtype) -> torch.Tensor: """Convert bool mask to TE float bias: visible→0.0, masked→-inf.""" # mask=True → masked → -inf; mask=False → visible → 0.0 return torch.zeros_like(mask, dtype=dtype).masked_fill_(mask, float('-inf'))

5. CUDA Backend（TE bias）— 复用 mask + 转换

prefix_sharing/backends/cuda_flash.py ★ 新建 (~80 行)
"""CUDA Flash Attention backend via Transformer Engine. Shares build_block_causal_mask with NPU backend, converts bool mask to TE float bias. """ from __future__ import annotations from typing import Any import torch from prefix_sharing.backends.base import BackendCapabilities from prefix_sharing.backends.torch_ref import TorchReferenceBackend from prefix_sharing.backends.mask import build_block_causal_mask, mask_to_te_bias from prefix_sharing.core.planner import PrefixSharingPlan class CudaFlashBackend(TorchReferenceBackend): capabilities = BackendCapabilities( name="cuda_flash", supports_cpu=False, supports_cuda=True, supports_cann=False, supports_different_q_kv_lengths=True, supports_prefix_last_restore=True, ) # build_kv() 完全继承 TorchReferenceBackend def attention( self, query, key, value, plan: PrefixSharingPlan, **kwargs ) -> Any: import transformer_engine as te num_heads = query.shape[1] head_dim = query.shape[2] scale = head_dim ** -0.5 # Step 1: 统一构造 bool mask（与 NPU 相同） bool_mask = build_block_causal_mask(plan, device=query.device) # Step 2: 转换为 TE float bias bias = mask_to_te_bias(bool_mask, dtype=query.dtype) # bias: [max_q, max_kv], 0.0=visible, -inf=masked # Step 3: cu_seqlens tensors cu_q = torch.tensor(plan.cu_seqlens_q, dtype=torch.int32, device=query.device) cu_kv = torch.tensor(plan.cu_seqlens_kv, dtype=torch.int32, device=key.device) # Step 4: 调用 TE te_attn = te.pytorch.DotProductAttention( num_attention_heads=num_heads, kv_channels=head_dim, attention_dropout=0.0, attn_mask_type='padding', softmax_scale=scale, ) output = te_attn( query.unsqueeze(1), key.unsqueeze(1), value.unsqueeze(1), attention_mask='padding', core_attention_bias_type='post_scale_bias', core_attention_bias=bias, qkv_format='thd', cu_seqlens_q=cu_q, cu_seqlens_k=cu_kv, max_seqlen_q=plan.max_seqlen_q, max_seqlen_k=plan.max_seqlen_kv, ) return output.squeeze(1)

6. Backend 工厂：自动选择平台

prefix_sharing/backends/cuda_ref.py (改写为工厂)
"""Backend factory: auto-detect platform and return the right backend.""" from __future__ import annotations def create_backend(name: str = "auto"): """Create the best available backend for the current platform. name options: "auto" | "npu_flash" | "cuda_flash" | "torch_ref" "auto" picks the fastest available kernel. """ if name in ("auto", "npu_flash"): try: from prefix_sharing.backends.npu_flash import NpuFlashBackend return NpuFlashBackend() except ImportError: pass if name in ("auto", "cuda_flash"): try: from prefix_sharing.backends.cuda_flash import CudaFlashBackend return CudaFlashBackend() except ImportError: pass # Fallback: pure PyTorch (CPU/CUDA/NPU any device) from prefix_sharing.backends.torch_ref import TorchReferenceBackend return TorchReferenceBackend()

8. 文件清单

核心设计：三个 backend 共享 build_block_causal_mask() 和 build_kv()（继承 torch_ref）。差异只在 attention() 方法中： mask 格式转换（NPU 直接传 bool，CUDA 转 float bias，Fallback 用 masked_fill）和调用不同的底层 kernel。新增代码共约 240 行，torch_ref 不改，始终作为 fallback。
