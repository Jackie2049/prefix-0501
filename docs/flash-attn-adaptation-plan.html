2. 逐文件详细修改

2.1 新建 backends/flash_attn.py

这是核心新文件，实现 FlashAttentionBackend，包含 3 个新方法。

prefix_sharing/backends/flash_attn.py ★ 新建文件
"""Flash Attention backend for prefix-sharing. Reuses TorchReferenceBackend.build_kv() for KV expansion, then pads Reuser Q tokens to match KV length and calls flash_attn_varlen_func with causal=True. """ from __future__ import annotations from typing import Any from prefix_sharing.backends.base import BackendCapabilities from prefix_sharing.backends.torch_ref import ( TorchReferenceBackend, _split_packed, ) from prefix_sharing.core.planner import PrefixSharingPlan class FlashAttentionBackend(TorchReferenceBackend): capabilities = BackendCapabilities( name="flash_attn", supports_cpu=False, supports_cuda=True, supports_cann=False, supports_different_q_kv_lengths=False, # pad 后 Q=KV 等长 supports_prefix_last_restore=True, ) # build_kv() 完全继承 TorchReferenceBackend，不 override # Provider 存 KV，Reuser torch.cat 拼接 — 逻辑不变 def pad_query( self, query: Any, plan: PrefixSharingPlan, ) -> tuple[Any, list[int]]: """Pad Reuser Q with zeros so Q length == KV length. Returns (padded_query, padded_cu_seqlens_q). """ torch = _torch() rows = _split_packed(query, plan.kept_lengths_q) padded_rows = [] padded_lengths = [] for i, row in enumerate(rows): if plan.is_reuser(i): prefix_len = plan.prefix_lens[i] pad_shape = (prefix_len,) + row.shape[1:] zeros = torch.zeros(pad_shape, dtype=row.dtype, device=row.device) padded_rows.append(torch.cat([zeros, row], dim=0)) padded_lengths.append(prefix_len + row.shape[0]) else: padded_rows.append(row) padded_lengths.append(row.shape[0]) return torch.cat(padded_rows, dim=0), padded_lengths def trim_output( self, output: Any, plan: PrefixSharingPlan, ) -> Any: """Remove padded positions from Flash Attention output. Reuser rows: drop prefix_len positions from the front. Provider rows: keep all positions unchanged. """ torch = _torch() padded_lengths = plan.expanded_lengths_kv # pad 后 Q 的长度 = KV 长度 rows = _split_packed(output, padded_lengths) trimmed = [] for i, row in enumerate(rows): if plan.is_reuser(i): offset = plan.prefix_lens[i] trimmed.append(row[offset:]) # 丢弃 pad 部分 else: trimmed.append(row) return torch.cat(trimmed, dim=0) def attention( self, query: Any, key: Any, value: Any, plan: PrefixSharingPlan, **_: Any, ) -> Any: from flash_attn import flash_attn_varlen_func torch = _torch() # Step 1: Pad Q padded_q, padded_lengths = self.pad_query(query, plan) # Step 2: Build cu_seqlens (Q 和 KV 等长) cu_seqlens = torch.tensor( [0] + list(_cumsum(padded_lengths)), dtype=torch.int32, device=padded_q.device, ) max_seqlen = max(padded_lengths) if padded_lengths else 0 # Step 3: Flash Attention (causal=True, Q=KV 等长) output = flash_attn_varlen_func( q=padded_q.unsqueeze(1), # [total, 1, heads, dim] k=key.unsqueeze(1), v=value.unsqueeze(1), cu_seqlens_q=cu_seqlens, cu_seqlens_k=cu_seqlens, # 同一个，Q=KV 等长 max_seqlen_q=max_seqlen, max_seqlen_k=max_seqlen, causal=True, ).squeeze(1) # [total, heads, dim] # Step 4: Trim padded positions return self.trim_output(output, plan) def _cumsum(lengths): total = 0 for l in lengths: total += l yield total
约 90 行。build_kv() 完全继承 TorchReferenceBackend，只新增 pad_query / trim_output / attention 三个方法

def maybe_run_prefix_sharing_attention(attention_module, query, key, value, ...): ctx = current_prefix_sharing_context() if ctx is None: return None # ... RoPE 处理不变 ... backend = ctx.backend # ★ backend 可能是 FlashAttentionBackend # ★ build_kv() 在两个 backend 中行为相同（继承） expanded_key, expanded_value = backend.build_kv( key, value, ctx.store, ctx.prefix_sharing_plan, layer_id=layer_id, tp_rank=tp_rank, ) # ★ attention() 内部自动处理 pad / FA / trim core_attn_out = backend.attention( query, expanded_key, expanded_value, ctx.prefix_sharing_plan, attention_mask=attention_mask, ) core_attn_out = core_attn_out.reshape(core_attn_out.size(0), 1, -1) return attention_module.linear_proj(core_attn_out)
关键：megatron_runtime.py 本身逻辑几乎不变！build_kv() 和 attention() 的调用签名完全相同。差别只在 backend 实例不同——FlashAttentionBackend.attention() 内部多了 pad/FA/trim 三步。
